  <action>
    Create src/app/api/sessions/[id]/corrections/validate-reason/route.ts.

    This route NEVER touches the database. It only calls the AI service.

    Implementation:
    ```typescript
    import { NextResponse } from "next/server";
    import { auth } from "@/lib/auth/config";
    import { validateReasonSchema } from "@/lib/validations/correction";
    import { validateCorrectionReason } from "@/lib/ai/service";
    import { ZodError } from "zod";

    export async function POST(request: Request) {
      const session = await auth();
      if (!session?.user) {
        return NextResponse.json({ error: "Not authenticated" }, { status: 401 });
      }

      let data;
      try {
        const body = await request.json();
        data = validateReasonSchema.parse(body);
      } catch (error) {
        if (error instanceof ZodError) {
          return NextResponse.json({ error: error.errors[0].message }, { status: 400 });
        }
        return NextResponse.json({ error: "Invalid request body" }, { status: 400 });
      }

      try {
        const result = await validateCorrectionReason(
          data.reason,
          session.user.preferredLanguage ?? undefined
        );
        return NextResponse.json(result);
      } catch {
        // AI is unavailable — return degraded neutral pass so UI does not block correction
        return NextResponse.json({ pass: true, feedback: null }, { status: 200 });
      }
    }
    ```

    This route:
    - Requires auth (401 if missing)
    - Validates reason length 20-500 chars via validateReasonSchema (400 if invalid)
    - Does NOT require session/series authorization — reason text is not sensitive
    - Does NOT perform any DB write
    - On AI failure: returns { pass: true, feedback: null } with 200 (graceful degradation)
    - Uses instanceof ZodError — NOT error.name === "ZodError"
