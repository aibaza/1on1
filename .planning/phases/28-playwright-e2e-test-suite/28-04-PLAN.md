    ```typescript
    import { test, expect } from "./fixtures";

    test.describe("Templates CRUD", () => {
      test.setTimeout(60_000);

      const TEMPLATE_NAME = `E2E Templates Test ${Date.now()}`;

      test("templates list loads with seeded template visible", async ({ adminPage }) => {
        await adminPage.goto("/templates");
        await expect(adminPage.getByRole("heading", { name: /templates/i })).toBeVisible({ timeout: 10_000 });
        // Seeded template exists
        await expect(adminPage.getByRole("link").first()).toBeVisible({ timeout: 10_000 });
        // Create Template button visible for admin
        await expect(adminPage.getByRole("button", { name: /create template/i })).toBeVisible();
      });

      test("admin can create a new template", async ({ adminPage }) => {
        await adminPage.goto("/templates");
        await adminPage.getByRole("button", { name: /create template/i }).first().click();
        await expect(adminPage.getByRole("dialog")).toBeVisible({ timeout: 5_000 });
        await adminPage.getByLabel(/template name/i).fill(TEMPLATE_NAME);
        await adminPage.getByLabel(/description/i).fill("Created by E2E test suite");
        // Submit the dialog — use the last button matching "create template" (inside the dialog)
        await adminPage.getByRole("button", { name: /create template/i }).last().click();
        // Template should appear in the list (redirect to /templates or name visible)
        await expect(adminPage.getByText(TEMPLATE_NAME)).toBeVisible({ timeout: 10_000 });
      });

      test("admin can navigate to template detail and add a question", async ({ adminPage }) => {
        await adminPage.goto("/templates");
        // Click into the newly created template or seeded template
        await adminPage.getByText(TEMPLATE_NAME).click({ timeout: 5_000 }).catch(async () => {
          // Fallback: click the first template in the list
          await adminPage.getByRole("link", { name: /template/i }).first().click();
        });
        await adminPage.waitForURL(/\/templates\//, { timeout: 10_000 });

        // Add a section if needed
        const addSectionBtn = adminPage.getByRole("button", { name: /add section/i });
        if (await addSectionBtn.isVisible({ timeout: 3_000 }).catch(() => false)) {
          await addSectionBtn.click();
          await adminPage.waitForTimeout(500);
        }

        // Add a question
        const addQuestionBtn = adminPage.getByRole("button", { name: /add question/i }).first();
        await addQuestionBtn.click();
        await expect(adminPage.getByRole("dialog")).toBeVisible({ timeout: 5_000 });

        // Fill question text
        const questionInput = adminPage.getByLabel(/question text/i);
        await questionInput.fill("E2E test question: how productive was this sprint?");
        // Submit the question dialog
        await adminPage.getByRole("button", { name: /^add question$/i }).click();
        await adminPage.waitForTimeout(1_000);

        // Question should appear in the template
        await expect(adminPage.getByText(/how productive was this sprint/i)).toBeVisible({ timeout: 10_000 });
      });

      test("admin can delete a template", async ({ adminPage }) => {
        // Create a template specifically for deletion
        const DELETE_TEMPLATE_NAME = `Delete Me E2E ${Date.now()}`;
        await adminPage.goto("/templates");
        await adminPage.getByRole("button", { name: /create template/i }).first().click();
        await adminPage.getByLabel(/template name/i).fill(DELETE_TEMPLATE_NAME);
        await adminPage.getByRole("button", { name: /create template/i }).last().click();
        await expect(adminPage.getByText(DELETE_TEMPLATE_NAME)).toBeVisible({ timeout: 10_000 });

        // Navigate into the template detail to find delete action
        await adminPage.getByText(DELETE_TEMPLATE_NAME).click();
        await adminPage.waitForURL(/\/templates\//, { timeout: 10_000 });

        // Look for Delete Template button (may be in an overflow menu or directly visible)
        const deleteBtn = adminPage.getByRole("button", { name: /delete template/i });
        if (await deleteBtn.isVisible({ timeout: 3_000 }).catch(() => false)) {
          await deleteBtn.click();
        } else {
          // Try overflow menu: button with "..." or "More options"
          const overflowBtn = adminPage.getByRole("button", { name: /more options|actions|\.\.\./i }).first();
          if (await overflowBtn.isVisible({ timeout: 3_000 }).catch(() => false)) {
            await overflowBtn.click();
            await adminPage.getByRole("menuitem", { name: /delete/i }).click({ timeout: 3_000 });
          }
        }

        // Confirm deletion dialog if present
        const confirmBtn = adminPage.getByRole("button", { name: /confirm|yes.*delete|delete/i }).last();
        if (await confirmBtn.isVisible({ timeout: 3_000 }).catch(() => false)) {
          await confirmBtn.click();
        }

        // Should redirect to templates list and template should not appear
        await adminPage.waitForURL(/\/templates$/, { timeout: 10_000 }).catch(() => {});
        await adminPage.goto("/templates"); // Ensure we're on the list
        await expect(adminPage.getByText(DELETE_TEMPLATE_NAME)).not.toBeVisible({ timeout: 5_000 });
      });
    });
    ```
