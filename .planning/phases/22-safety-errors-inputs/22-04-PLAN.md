```tsx
import { getTranslations } from "next-intl/server";
import Link from "next/link";
import { Button } from "@/components/ui/button";

export default async function SessionNotFound() {
  const t = await getTranslations("sessions");
  return (
    <div className="flex flex-col items-center justify-center py-24 text-center gap-4">
      <h1 className="text-2xl font-semibold">{t("notFound.title")}</h1>
      <p className="text-muted-foreground">{t("notFound.description")}</p>
      <Button asChild variant="outline">
        <Link href="/history">{t("notFound.backToSessions")}</Link>
      </Button>
    </div>
  );
}
```

- Export is `default` (Next.js requires this for not-found.tsx)
- No `"use client"` directive — Server Component
- No manual layout wrapper — the (dashboard) layout is inherited automatically
- Link goes to `/history` (the history/sessions list page in the project)
- Do NOT place any not-found.tsx at sessions/[id]/ — only at sessions/[id]/summary/
